-- Skybox check
local function matchesSkybox(sky, target)
    for key, id in pairs(target) do
        local val = sky[key]
        if not val or string.lower(val) ~= string.lower(id) then
            return false
        end
    end
    return true
end

local function isBloodmoon()
    local sky = Lighting:FindFirstChildWhichIsA("Sky")
    return sky and matchesSkybox(sky, bloodlitSkyboxIds)
end

-- Improved server list fetcher
local function getServers()
    local servers = {}
    local cursor = ""
    local baseUrl = "https://games.roblox.com/v1/games/%s/servers/Public?sortOrder=Asc&limit=100"
    for _ = 1, 5 do
        local url = baseUrl:format(PlaceId)
        if cursor ~= "" then
            url = url .. "&cursor=" .. cursor
        end
        local success, response = pcall(function()
            return HttpService:JSONDecode(game:HttpGet(url))
        end)
        if success and response and response.data then
            for _, server in ipairs(response.data) do
                if server.id ~= game.JobId and not server.vipServerId and server.playing < server.maxPlayers then
                    table.insert(servers, server)
                end
            end
            if response.nextPageCursor then
                cursor = response.nextPageCursor
            else
                break
            end
        else
            break
        end
    end
    return servers
end
